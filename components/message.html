Message 消息提示 ​

常用的 Message

基础用法 ​

基础 Message. 支持多种方式调用

<script lang="ts" setup>
import { h } from "vue";
import { MyMessage } from "@qiwen72/my-component";

const open = () => {
  MyMessage("This is a message.");
};

const openVn = () => {
  MyMessage({
    message: h("p", { style: "line-height: 1; font-size: 14px" }, [
      h("span", null, "Message can be "),
      h("i", { style: "color: teal" }, "VNode"),
    ]),
  });
};

const openInsert = () => {
  MyMessage({
    message: "hihi",
    type: "success",
  });
};

const openFn = () => {
  MyMessage.success("hihihihi");
};
</script>

<template>
  <my-button :plain="true" @click="open">string</my-button>
  <my-button :plain="true" @click="openVn">VNode</my-button>
  <my-button :plain="true" @click="openInsert">插件式调用</my-button>
  <my-button :plain="true" @click="openFn">函数式调用</my-button>
  <my-button :plain="true" @click="$message.success('hihihi')"
    >全局方法调用</my-button
  >
</template>

不同类型 ​

有四种不同的类型

<script lang="ts" setup>
import { MyMessage } from "@qiwen72/my-component";

const openFn1 = () => {
  MyMessage.success("hihihihi");
};
const openFn2 = () => {
  MyMessage.warning("hihihihi");
};
const openFn3 = () => {
  MyMessage.error("hihihihi");
};
const openFn4 = () => {
  MyMessage.info("hihihihi");
};
</script>

<template>
  <my-button :plain="true" @click="openFn1">success</my-button>
  <my-button :plain="true" @click="openFn2">warning</my-button>
  <my-button :plain="true" @click="openFn3">error</my-button>
  <my-button :plain="true" @click="openFn4">info</my-button>
</template>

可关闭的 ​

使用 showClose属性来定义是否能关闭

<script lang="ts" setup>
import { MyMessage } from "@qiwen72/my-component";

const openFn1 = () => {
  MyMessage.success({ message: "hihihihi", showClose: true });
};
const openFn2 = () => {
  MyMessage.warning({ message: "hihihihi", showClose: true });
};
const openFn3 = () => {
  MyMessage.error({ message: "hihihihi", showClose: true });
};
const openFn4 = () => {
  MyMessage.info({ message: "hihihihi", showClose: true });
};
</script>

<template>
  <my-button :plain="true" @click="openFn1">success</my-button>
  <my-button :plain="true" @click="openFn2">warning</my-button>
  <my-button :plain="true" @click="openFn3">error</my-button>
  <my-button :plain="true" @click="openFn4">info</my-button>
</template>

文字居中 ​

可以通过 center 属性来设置文字居中

<script lang="ts" setup>
import { MyMessage } from "@qiwen72/my-component";

const openFn1 = () => {
  MyMessage.success({
    message: "hihihih center",
    showClose: true,
    center: true,
  });
};
</script>

<template>
  <my-button :plain="true" @click="openFn1">center</my-button>
</template>

CloseAll ​

可以通过 $message.closeAll 一键全部关闭.

<script lang="ts" setup>
import { MyMessage } from "@qiwen72/my-component";

const openFn1 = () => {
  MyMessage.success({
    message: "hihihih center",
    showClose: true,
    center: true,
  });
};
</script>

<template>
  <my-button type="primary" @click="$message.closeAll()">close All</my-button>
  <my-button :plain="true" @click="openFn1">center</my-button>
</template>
