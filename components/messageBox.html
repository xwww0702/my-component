messageBox 消息弹出框 ​

常用的 messageBox

基础用法 ​

基础 messageBox. 有三种不同的类型，alert、confirm和prompt

<script setup lang="ts">
import { MyMessageBox, MyMessage } from "@qiwen72/my-component";

function openAlert() {
  MyMessageBox.alert("This is a message", "Title")
    .then((action) => {
      MyMessage.info(`action: ${action}`);
    })
    .catch((action) => {
      MyMessage.warning(`action: ${action}`);
    });
}

function openConfirm() {
  MyMessageBox.confirm(
    "proxy will permanently delete the file. Continue?",
    "Warning",
    { type: "warning" }
  )
    .then((action) => {
      MyMessage.info(`action: ${action}`);
    })
    .catch((action) => {
      MyMessage.warning(`action: ${action}`);
    });
}

function openPrompt() {
  MyMessageBox.prompt("请输入你的年龄", "Tip", { type: "info" })
    .then((value) => {
      MyMessage.info(`你的年龄: ${value}`);
    })
    .catch((action) => {
      MyMessage.warning(`action: ${action}`);
    });
}
</script>

<template>
  <my-button @click="openAlert" plain> Click to open the Alert</my-button>
  <my-button @click="openConfirm" plain> Click to open the Confirm</my-button>
  <my-button @click="openPrompt" plain> Click to open the Prompt</my-button>
</template>

使用 VNode 传入内容 ​

<script setup lang="ts">
import { MyMessageBox } from "@qiwen72/my-component";
import { h } from "vue";
function openAlert() {
  MyMessageBox({
    title: "Message",
    message: h("p", null, [
      h("span", null, "Message can be "),
      h("i", { style: "color: teal" }, "VNode"),
    ]),
  });
}
</script>

<template>
  <my-button @click="openAlert" plain> Click to open the Vnode Box</my-button>
</template>

可关闭的 ​

使用 showClose属性来定义是否显示关闭 icon

<script setup lang="ts">
import { MyMessageBox, MyMessage } from "@qiwen72/my-component";

function openAlert() {
  MyMessageBox.alert("This is a message", "Title")
    .then((action) => {
      MyMessage.info(`action: ${action}`);
    })
    .catch((action) => {
      MyMessage.warning(`action: ${action}`);
    });
}

function openConfirm() {
  MyMessageBox.confirm(
    "proxy will permanently delete the file. Continue?",
    "Warning",
    { type: "warning", showClose: false }
  )
    .then((action) => {
      MyMessage.info(`action: ${action}`);
    })
    .catch((action) => {
      MyMessage.warning(`action: ${action}`);
    });
}
</script>

<template>
  <my-button @click="openAlert" plain> showClose true</my-button>
  <my-button @click="openConfirm" plain>showClose false</my-button>
</template>

点击外部关闭 ​

可以通过 closeOnClickModal 来设置是否能点击外部关闭

<script setup lang="ts">
import { MyMessageBox, MyMessage } from "@qiwen72/my-component";

function openAlert() {
  MyMessageBox.confirm("This is a message", "Title", {
    type: "warning",
    closeOnClickModal: false,
  })
    .then((action) => {
      MyMessage.info(`action: ${action}`);
    })
    .catch((action) => {
      MyMessage.warning(`action: ${action}`);
    });
}

function openConfirm() {
  MyMessageBox.confirm(
    "proxy will permanently delete the file. Continue?",
    "Warning",
    { type: "warning", closeOnClickModal: true }
  )
    .then((action) => {
      MyMessage.info(`action: ${action}`);
    })
    .catch((action) => {
      MyMessage.warning(`action: ${action}`);
    });
}
</script>

<template>
  <my-button @click="openAlert" plain> closeOnClickModal false</my-button>
  <my-button @click="openConfirm" plain>closeOnClickModal true</my-button>
</template>

自定义 ​

可以自定义 MessageBox，并且可以利用钩子函数在提交或取消时执行回调函数。

<script setup lang="ts">
import { MyMessageBox, MyMessage } from "@qiwen72/my-component";
import { delay } from "lodash-es";

async function openAlert() {
  try {
    const action = await MyMessageBox({
      title: "My Custom Box",
      message: "hihihi",
      showCancelButton: true,
      confirmButtonText: "Yes",
      cancelButtonText: "No",
      type: "danger",
      icon: "spin",
      beforeClose(action, instance, done) {
        if (action !== "confirm") {
          done();
          return;
        }

        instance.confirmButtonLoading = true;
        instance.confirmButtonText = "Loading...";
        delay(() => {
          done();
          delay(() => (instance.confirmButtonLoading = false), 1000);
        }, 3000);
      },
    });
    MyMessage.success("success");
  } catch (action) {
    MyMessage.warning(`action:${action}`);
  }
}
</script>

<template>
  <my-button @click="openAlert" plain> click me</my-button>
</template>
